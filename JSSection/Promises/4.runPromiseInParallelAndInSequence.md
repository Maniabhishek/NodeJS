```js
function runInSequence(tasks) {
    return tasks.reduce((promiseChain, task) => {
      return promiseChain.then(results =>
        task().then(result => [...results, result])
      );
    }, Promise.resolve([]));
  }
 ```

```js
function runInParallel(tasks) {
  return Promise.all(tasks.map(task => task()));
}
```
