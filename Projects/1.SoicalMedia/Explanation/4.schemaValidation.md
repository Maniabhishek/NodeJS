```js
const mongoose = require("mongoose")

const userSchema = new mongoose.Schema({
    firstName: {
        type: String,
        required: true
    },
    lastName: {
        type: String
    },
    email: {
        type: String,
        required: true,
        unique: true,
        trim: true
    },
    age: {
        type: Number,
        min: 18,
        max: 60
    },
    gender: {
        type: String,
        lowercase: true,
        trim: true,
        enum: {
            values: ["male", "female", "others"],
            message: '{VALUE} is not supported' 
        }
    },
    about: {
        type: String,
        default: "This default of about",
    },
    photoUrl: {
        type: String,
        default: "abc.com",
    },
    skills: {
        type: [String],
        validate: {
            validator: (value) => {
                if(!value || !Array.isArray(value) || value.length > 10){
                    return false
                }
                return true
            }, 
        message: "skills not valid"
        }
    }
}, {
    timestamps: true
})

const UserModel = mongoose.model("User", userSchema)
module.exports = UserModel

```

- above are some of the schema level validation
- skills have custom validators we can either add validator and message or we can simply write a arrow function to validate function shown below
